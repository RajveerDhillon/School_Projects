View: Biggest Releases First in North America
Select 
    Video_Game.Title,
    Distribute."Release Date North America",
    MAX(Distribute.Copies_Sold)
From 
    Video_Game
    Left Join Distribute Using (vgID)
Where
    (Distribute."Release Date North America" < Distribute."Release Date Japan")


View: Games Where JP and NA Release Dates are the Same
Select 
    Video_Game.Title,
    Distribute."Release Date Japan",
    Distribute."Release Date North America"
From
    Video_Game
    Left Join Distribute Using (vgID)
Where Distribute."Release Date Japan" = Distribute."Release Date North America"


View: Highest Rated Game
Select 
    Video_Game.Title, MAX(Rating.MetaCritic)
from Video_Game
    Left Join Rating using (vgID)

View: Outsourced vs Internal Games Performance
Select 
    Video_Game.Title,
    Developers.Name,
    Square_Enix.Developers,
    Distribute.Copies_Sold
from 
    Video_Game
    Left Join Distribute using (vgID)
    Left Join Square_Enix Using (sqID)
    Left Join "Contract(Pub to Dev)" Using (sqID)
    Left Join Developers Using (devID)


View: Releases on Exclusive Platforms
Select
    Video_Game.Title,
    case 
        WHEN (Platform."Microsoft Windows" = 1 AND Platform.PS4 = 1 AND Platform."Xbox One" = 1) OR 
        (Platform."Microsoft Windows" = 1 AND Platform.PS4 = 1 AND Platform."Nintendo Switch" = 1) OR
        (Platform."Microsoft Windows" = 1 AND Platform."Xbox One" = 1 AND Platform."Nintendo Switch" = 1) OR
        (Platform.PS4 = 1 AND Platform."Xbox One" = 1 AND Platform."Nintendo Switch" = 1)
        THEN 'NO'
        else 'YES'
    end as PlatformExclusive,
    Platform."Microsoft Windows",
    Platform.PS4,
    Platform."Xbox One",
    Platform."Nintendo Switch"
From
    Video_Game
    Left Join Platform Using (vgID)

View: Same Director with Successful Games
Select Distinct
    Video_Game.Title,
    count(Director.Name),
    Distribute.Copies_Sold,
    Produce.vgID as p_vgId,
    Produce.dirID as p_dirID
from 
    Video_Game 
    Left Join Produce ON Video_Game.vgID = p_vgID
    Left Join Director ON Director.dirID = p_dirID
    Left Join Distribute ON Distribute.vgID = Video_Game.vgID
Group by Director.Name
Having count(Director.Name) > 1


View: Which Games have more than 2 Genres
Select
    Video_Game.Title,
    Genre.Gen1,
    Genre.Gen2
From
    Video_Game
    Left Join Genre Using (vgID)
Where Genre.Gen2 IS NOT NULL
